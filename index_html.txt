  <script>
    // auto year
    document.getElementById('year').textContent = new Date().getFullYear();

    // Lightbox logic
    const cards = document.querySelectorAll('.card');
    const lightbox = document.getElementById('lightbox');
    const lightboxImg = document.getElementById('lightboxImg');
    const lightboxTitle = document.getElementById('lightboxTitle');
    const closeBtn = document.getElementById('closeBtn');

    cards.forEach(card => {
      card.addEventListener('click', () => {
        const img = card.querySelector('img');
        lightboxImg.src = img.src;
        lightboxImg.alt = img.alt;
        lightboxTitle.textContent = card.dataset.title || '';
        lightbox.classList.add('open');
        lightbox.setAttribute('aria-hidden', 'false');
      });
    });

    closeBtn.addEventListener('click', () => {
      lightbox.classList.remove('open');
      lightbox.setAttribute('aria-hidden', 'true');
      lightboxImg.src = '';
    });

    lightbox.addEventListener('click', (e) => {
      if (e.target === lightbox) {
        lightbox.classList.remove('open');
        lightboxImg.src = '';
        lightbox.setAttribute('aria-hidden','true');
      }
    });

    // Basic form UX (client-side)
    const contactForm = document.getElementById('contactForm');
    const contactStatus = document.getElementById('contactStatus');
    contactForm.addEventListener('submit', (e) => {
      // Let the form submit to Formspree — but show immediate feedback.
      contactStatus.textContent = 'Sending…';
      setTimeout(() => {
        contactStatus.textContent = 'If you used a demo endpoint, replace the form action with your Formspree URL. Otherwise check your email service.';
      }, 900);
    });

    const commissionForm = document.getElementById('commissionForm');
    const commissionStatus = document.getElementById('commissionStatus');
    commissionForm.addEventListener('submit', (e) => {
      commissionStatus.textContent = 'Sending…';
      setTimeout(() => {
        commissionStatus.textContent = 'If you want file uploads to work, connect to a backend (Formspree paid plan or Netlify Forms).';
      }, 900);
    });

    function scrollToContact(){
      document.getElementById('contactForm').scrollIntoView({behavior:'smooth', block:'center'});
    }

    // Accessibility: close lightbox with ESC
    document.addEventListener('keydown', (e) => {
      if (e.key === 'Escape' && lightbox.classList.contains('open')) {
        lightbox.classList.remove('open');
        lightboxImg.src = '';
        lightbox.setAttribute('aria-hidden','true');
      }
    });
  </script>
