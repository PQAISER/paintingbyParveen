  <script>
    // Auto year
    document.getElementById('year').textContent = new Date().getFullYear();

    // Lightbox
    const cards = document.querySelectorAll('.card');
    const lightbox = document.getElementById('lightbox');
    const lightboxImg = document.getElementById('lightboxImg');
    const lightboxTitle = document.getElementById('lightboxTitle');
    const closeBtn = document.getElementById('closeBtn');

    cards.forEach(card => {
      card.addEventListener('click', () => {
        const img = card.querySelector('img');
        lightboxImg.src = img.src;
        lightboxImg.alt = img.alt;
        lightboxTitle.textContent = card.dataset.title || '';
        lightbox.classList.add('open');
        lightbox.setAttribute('aria-hidden','false');
      });
    });

    closeBtn.addEventListener('click', ()=>{
      lightbox.classList.remove('open');
      lightbox.setAttribute('aria-hidden','true');
      lightboxImg.src='';
    });

    lightbox.addEventListener('click',(e)=>{
      if(e.target===lightbox){
        lightbox.classList.remove('open');
        lightboxImg.src='';
        lightbox.setAttribute('aria-hidden','true');
      }
    });

    document.addEventListener('keydown',(e)=>{
      if(e.key==='Escape' && lightbox.classList.contains('open')){
        lightbox.classList.remove('open');
        lightboxImg.src='';
        lightbox.setAttribute('aria-hidden','true');
      }
    });

    function scrollToContact(){
      document.getElementById('contactForm').scrollIntoView({behavior:'smooth', block:'center'});
    }

    // Formspree AJAX
    function handleFormAjax(formId,statusId){
      const form = document.getElementById(formId);
      const status = document.getElementById(statusId);
      form.addEventListener('submit', async (e)=>{
        e.preventDefault();
        status.textContent='Sending…';
        const formData = new FormData(form);
        try{
          const response = await fetch(form.action,{
            method:'POST',
            body:formData,
            headers:{'Accept':'application/json'}
          });
          if(response.ok){
            status.textContent='Thank you! Your message has been sent.';
            form.reset();
          } else {
            const data = await response.json();
            status.textContent=data?.errors?.[0]?.message || 'Oops! There was a problem sending your message.';
          }
        }catch(err){
          status.textContent='Network error. Please try again later.';
        }
      });
    }

    handleFormAjax('contactForm','contactStatus');
    handleFormAjax('commissionForm','commissionStatus');
  </script>
